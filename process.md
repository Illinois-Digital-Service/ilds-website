            <br/>
            <p>Example Outcomes:</p>
            <ul class="asterisk-list">
              {% include asterisk-bullet.html bg="var(--yellow-1)" fg="var(--green)" text="A redesigned web page becomes easier to navigate" %}

          </div>
        </div>

            <p>Example Outcomes:</p>
